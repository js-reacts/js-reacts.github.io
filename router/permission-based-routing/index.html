    Permission Based Routing

    To implement permission-based routing in the AppRoutes component you've shared, you will need to incorporate checks that will verify if the user has the necessary permissions to access certain routes. Given that you're using Redux for authentication, the approach would involve both authentication and permission checks.

    PrivateRoute.js
    import React from "react";
import { Navigate } from "react-router-dom";
import { useSelector } from "react-redux";

// Permission-based PrivateRoute component
const PrivateRoute = ({ children, requiredPermissions }) => {
  const isAuthenticated = useSelector((state) => state.auth.isAuthenticated);
  const userPermissions = useSelector((state) => state.auth.permissions); // Assuming permissions are stored here

  // If not authenticated, redirect to login page
  if (!isAuthenticated) {
    return <Navigate to="/login" />;
  }

  // If user doesn't have the required permissions, redirect to "no access" page
  if (
    requiredPermissions &&
    !requiredPermissions.every((permission) =>
      userPermissions.includes(permission)
    )
  ) {
    return <Navigate to="/no-access" />;
  }

  return children;
};

export default PrivateRoute;

    Hide the Menu

    To hide menu items in your Sidebar based on user permissions, you can follow these steps:

    • Get the user's permissions from the Redux store (assuming the permissions are already part of the user’s data).
    • Filter the menu items by checking whether the user has the necessary permissions before rendering the item.
    • Update the sidebar items dynamically to only show items that the user is authorized to view.

    Here's how you can modify your Sidebar component to achieve this:

    1. Get Permissions from Redux Store:

    You can use the useSelector hook to access the user's permissions from the Redux store. For this example, I'm assuming that the permissions are stored in state.auth.permissions.

    2. Filter the Menu Items:

    Before rendering the menu, check whether the user has the necessary permissions for each menu item. You can add a condition that checks if the user has the required permissions for each route.

    3. Update the Sidebar Component:

    import React, { useState, useEffect } from "react";
import { Menu } from "antd";
import {
  DashboardOutlined,
  SettingOutlined,
  UserOutlined,
  AccountBookOutlined,
  BankOutlined,
  FundOutlined,
  ProjectOutlined,
  IssuesCloseOutlined,
} from "@ant-design/icons";
import { useNavigate, useLocation } from "react-router-dom";
import { useSelector } from "react-redux";
import { APP_ROUTES } from "@shared/routes";
import logger from "@shared/utils/logger";

const Sidebar = ({ collapsed, height, width }) => {
  const navigate = useNavigate();
  const location = useLocation();
  const permissions = useSelector((state) => state.auth.permissions); // Assuming permissions are stored here

  // Define the menu items
  const items = [
    {
      key: APP_ROUTES.DASHBOARD,
      label: "Dashboard",
      icon: <DashboardOutlined />,
      children: [
        {
          key: APP_ROUTES.DASHBOARD_ANALYSIS,
          label: "Analysis",
          requiredPermissions: [
            "Permissions.Application.CanViewDataDuringBusinessHours",
          ],
        },
        {
          key: APP_ROUTES.DASHBOARD_WORKSPACE,
          label: "Workplace",
          requiredPermissions: [
            "Permissions.Application.CanViewDataWeekendOnly",
          ],
        },
      ],
    },
    {
      key: APP_ROUTES.FOREX,
      label: "Forex T Deal",
      icon: <AccountBookOutlined />,
      children: [
        {
          key: APP_ROUTES.FOREX_DEALS,
          label: "All Deals",
          requiredPermissions: ["Permissions.Application.CanViewDashboard"],
        },
        {
          key: APP_ROUTES.FOREX_TRADING,
          label: "Buying / Selling",
          requiredPermissions: ["Permissions.Application.CanViewDashboard"],
        },
      ],
    },
    {
      key: "/lcy",
      label: "M.M.D. - Lcy",
      icon: <BankOutlined />,
      children: [
        {
          key: "/lcy/option1",
          label: "Option 7",
        },
      ],
    },
    // ... other menu items (similarly)
  ];

  // Function to filter items based on permissions
  const filterItemsByPermissions = (items) => {
    return items.filter((item) => {
      // If item has children, filter the children recursively
      if (item.children) {
        item.children = filterItemsByPermissions(item.children);
      }

      // Only include the item if it has no required permissions or the user has them
      return (
        !item.requiredPermissions ||
        item.requiredPermissions.some((perm) => permissions.includes(perm))
      );
    });
  };

  // Filter menu items based on permissions
  const filteredItems = filterItemsByPermissions(items);

  // Function to get parent key for open submenu
  const getParentKey = (path) => {
    const parent = filteredItems.find((item) => path.startsWith(item.key));
    return parent ? parent.key : null;
  };

  const [openKeys, setOpenKeys] = useState([getParentKey(location.pathname)]);

  useEffect(() => {
    setOpenKeys([getParentKey(location.pathname)]);
  }, [location.pathname]);

  const handleClick = ({ key }) => {
    navigate(key);
  };

  return (
    <Menu
      className="sidebar-layout"
      style={{
        height: height,
        width: width,
        textAlign: "left",
      }}
      mode="inline"
      selectedKeys={[location.pathname]}
      inlineCollapsed={collapsed}
      openKeys={openKeys}
      onOpenChange={setOpenKeys}
      onClick={handleClick}
      items={filteredItems} // Use the filtered items
    />
  );
};

export default Sidebar;
    Key Changes

    1. Permissions Check:

      • Each item now has an optional requiredPermissions field. If it has this field, it will be an array of permissions required for that item to be visible.
      • The filterItemsByPermissions function filters out items that the user doesn't have permissions for.

    2. Recursive Filtering:

      • Since some items have children, the filterItemsByPermissions function filters both parent and child items recursively.
      • If an item or its children don't match the permissions, they will be excluded from the menu.

    3. Dynamic Menu Rendering:

      • The filteredItems array is used to render the menu, ensuring that only items the user is authorized to see are included.

    4. Permissions Configuration:

    Make sure that you have permissions set up properly in your Redux store and that each route/component has its necessary permission defined (like Permissions.Application.CanViewDashboard, etc.).

    5. Example:

    • If a user doesn't have the Permissions.Application.CanViewDataDuringBusinessHours, the "Analysis" link will not be displayed in the sidebar.
    • Similarly, any other menu item that requires specific permissions will be hidden from the sidebar if the user doesn't have the necessary permissions.

    By following this approach, you ensure that menu items are not just hidden but also prevented from rendering based on the user's permissions, making the app's navigation more secure and dynamic.

    Hiding the Parent as Well is not children

    Great question! In the scenario where both the "All Deals" and "Buying / Selling" menu items under "Forex T Deal" do not have the canView permission, the "Forex T Deal" text (parent label) would still be shown unless you explicitly filter out the parent item as well.

    Currently, in the provided code, the filtering logic only checks the requiredPermissions on the children. If none of the children have permission, the parent (in this case, "Forex T Deal") will still be visible, even if all the child items are hidden.

    To address this, you can modify the filtering logic to also check if the parent item itself should be shown based on the visibility of its children. If none of the child items are allowed to be shown due to missing permissions, you can hide the entire parent as well.

    Here’s how to fix this:

    You need to update the filtering function so that it also hides the parent item if none of its children are visible (based on permissions).

    import React, { useState, useEffect } from "react";
import { Menu } from "antd";
import {
  DashboardOutlined,
  SettingOutlined,
  AccountBookOutlined,
  BankOutlined,
  FundOutlined,
  ProjectOutlined,
  IssuesCloseOutlined,
} from "@ant-design/icons";
import { useNavigate, useLocation } from "react-router-dom";
import { useSelector } from "react-redux";
import { APP_ROUTES } from "@shared/routes";
import logger from "@shared/utils/logger";

const Sidebar = ({ collapsed, height, width }) => {
  const navigate = useNavigate();
  const location = useLocation();
  const permissions = useSelector((state) => state.auth.permissions); // Assuming permissions are stored here

  // Define the menu items
  const items = [
    {
      key: APP_ROUTES.DASHBOARD,
      label: "Dashboard",
      icon: <DashboardOutlined />,
      children: [
        {
          key: APP_ROUTES.DASHBOARD_ANALYSIS,
          label: "Analysis",
          requiredPermissions: [
            "Permissions.Application.CanViewDataDuringBusinessHours",
          ],
        },
        {
          key: APP_ROUTES.DASHBOARD_WORKSPACE,
          label: "Workplace",
          requiredPermissions: [
            "Permissions.Application.CanViewDataWeekendOnly",
          ],
        },
      ],
    },
    {
      key: APP_ROUTES.FOREX,
      label: "Forex T Deal",
      icon: <AccountBookOutlined />,
      children: [
        {
          key: APP_ROUTES.FOREX_DEALS,
          label: "All Deals",
          requiredPermissions: ["Permissions.Application.CanViewDashboard"],
        },
        {
          key: APP_ROUTES.FOREX_TRADING,
          label: "Buying / Selling",
          requiredPermissions: ["Permissions.Application.CanViewDashboard"],
        },
      ],
    },
    // Other items...
  ];

  // Function to filter items based on permissions
  const filterItemsByPermissions = (items) => {
    return items.filter((item) => {
      // If item has children, filter the children recursively
      if (item.children) {
        item.children = filterItemsByPermissions(item.children);

        // If no child is visible, exclude the parent item as well
        if (item.children.length === 0) {
          return false;
        }
      }

      // Only include the item if it has no required permissions or the user has them
      return (
        !item.requiredPermissions ||
        item.requiredPermissions.some((perm) => permissions.includes(perm)) ||
        (item.children && item.children.length > 0) // Keep parent if it has visible children
      );
    });
  };

  // Filter menu items based on permissions
  const filteredItems = filterItemsByPermissions(items);

  // Function to get parent key for open submenu
  const getParentKey = (path) => {
    const parent = filteredItems.find((item) => path.startsWith(item.key));
    return parent ? parent.key : null;
  };

  const [openKeys, setOpenKeys] = useState([getParentKey(location.pathname)]);

  useEffect(() => {
    setOpenKeys([getParentKey(location.pathname)]);
  }, [location.pathname]);

  const handleClick = ({ key }) => {
    navigate(key);
  };

  return (
    <Menu
      className="sidebar-layout"
      style={{
        height: height,
        width: width,
        textAlign: "left",
      }}
      mode="inline"
      selectedKeys={[location.pathname]}
      inlineCollapsed={collapsed}
      openKeys={openKeys}
      onOpenChange={setOpenKeys}
      onClick={handleClick}
      items={filteredItems} // Use the filtered items
    />
  );
};

export default Sidebar;
    Key Changes

    1. Filtering Parents Based on Children Visibility:

      • In the filterItemsByPermissions function, after filtering the child items, we check if any children are visible. If none of the children are visible, we return false to prevent the parent item from being included.
      • The parent menu item is only included if it has visible children or if the parent itself is permitted.

    2. Excluding the Parent if No Child Has Permission:

      • In the filtering logic, if none of the children for a parent item are visible due to permission constraints, the parent item (e.g., "Forex T Deal") will be excluded as well.
    Example
    • If neither "All Deals" nor "Buying / Selling" has the necessary permissions, the entire "Forex T Deal" section (parent item) will be hidden from the sidebar.
    • If one of the child items under "Forex T Deal" has permissions, the parent item ("Forex T Deal") will be visible with the corresponding child item(s) displayed.

    By applying this logic, your sidebar dynamically adjusts to the user's permissions, ensuring that both parent and child menu items are conditionally rendered.

      children: [
        {
          key: APP_ROUTES.FOREX_DEALS,
          label: "All Deals",
          requiredPermissions: [Permissions.CAN_VIEW_FOREX_DEALS],
        },
        {
          key: APP_ROUTES.FOREX_TRADING,
          label: "Buying / Selling",
          requiredPermissions: [Permissions.CAN_VIEW_FOREX_TRADING],
        },
      ],
    },
    // Other items...
  ];

  // Function to filter items based on permissions
  const filterItemsByPermissions = (items) => {
    return items.filter((item) => {
      // If item has children, filter the children recursively
      if (item.children) {
        item.children = filterItemsByPermissions(item.children);

        // If no child is visible, exclude the parent item as well
        if (item.children.length === 0) {
          return false;
        }
      }

      // Only include the item if it has no required permissions or the user has them
      return (
        !item.requiredPermissions ||
        item.requiredPermissions.some((perm) => permissions.includes(perm)) ||
        (item.children && item.children.length > 0) // Keep parent if it has visible children
      );
    });
  };

  // Filter menu items based on permissions
  const filteredItems = filterItemsByPermissions(items);

  // Function to get parent key for open submenu
  const getParentKey = (path) => {
    const parent = filteredItems.find((item) => path.startsWith(item.key));
    return parent ? parent.key : null;
  };

  const [openKeys, setOpenKeys] = useState([getParentKey(location.pathname)]);

  useEffect(() => {
    setOpenKeys([getParentKey(location.pathname)]);
  }, [location.pathname]);

  const handleClick = ({ key }) => {
    navigate(key);
  };

  return (
    <Menu
      className="sidebar-layout"
      style={{
        height: height,
        width: width,
        textAlign: "left",
      }}
      mode="inline"
      selectedKeys={[location.pathname]}
      inlineCollapsed={collapsed}
      openKeys={openKeys}
      onOpenChange={setOpenKeys}
      onClick={handleClick}
      items={filteredItems} // Use the filtered items
    />
  );
};

export default Sidebar;

    1. Permissions Constants:

      • I imported the Permissions constants from @constants/permissions.js.
      • The permissions used in each item are now referenced from the Permissions constants, e.g., Permissions.CAN_VIEW_DATA_DURING_BUSINESS_HOURS, Permissions.CAN_VIEW_FOREX_DEALS, etc.

    2. Easier Maintenance:

      • By centralizing your permissions in a constants file, you reduce the risk of hardcoding strings multiple times, and you make it easier to update permissions in the future. If a permission key changes, you only need to update it in one place (the permissions.js file).
    • Centralized Management: All permission keys are now in one place, making it easier to update and manage.
    • Cleaner Code: Instead of repeating long permission strings, you use meaningful constants.
    • Consistency: Reduces the chances of typos or errors in permission strings.

    3. Optional: Use Permissions in Other Components

    You can similarly import the Permissions constants in other parts of your application where you need to check permissions, making your code consistent and modular.

    import { Permissions } from "@constants/permissions"; // Import permissions constants

// Example of how you could use them
if (userPermissions.includes(Permissions.CAN_VIEW_DASHBOARD)) {
  // Grant access to the dashboard
}

    This pattern keeps your permission checks clean and easy to manage across the app.

    Role-based or Permission-based Routing

    Role-based or permission-based routing is an extension of protected routing where routes are conditionally rendered based on the user’s permissions or roles. Permissions can be fetched from the server, or you can store them in global state (e.g., Redux, Context API) after a successful login.

    This is the type of routing you've asked about: if a user doesn’t have the required permissions, the route won’t be shown or accessed.

    Example of Permission-based Routing
    import React from "react";
import { BrowserRouter as Router, Route, Redirect } from "react-router-dom";

// Mock permission data
const permissions = ["CAN_VIEW_DASHBOARD"];

const hasPermission = (permission) => permissions.includes(permission);

const Dashboard = () => <div>Dashboard</div>;
const Settings = () => <div>Settings</div>;

const ProtectedRoute = ({
  component: Component,
  requiredPermission,
  ...rest
}) => {
  return (
    <Route
      {...rest}
      render={(props) =>
        hasPermission(requiredPermission) ? (
          <Component {...props} />
        ) : (
          <Redirect to="/no-permission" />
        )
      }
    />
  );
};

const NoPermission = () => (
  <div>You do not have permission to view this page.</div>
);

const App = () => {
  return (
    <Router>
      <Switch>
        <Route path="/no-permission" component={NoPermission} />
        <ProtectedRoute
          path="/dashboard"
          component={Dashboard}
          requiredPermission="CAN_VIEW_DASHBOARD"
        />
        <ProtectedRoute
          path="/settings"
          component={Settings}
          requiredPermission="CAN_VIEW_SETTINGS"
        />
      </Switch>
    </Router>
  );
};

export default App;

    In this example, the ProtectedRoute component checks whether the user has the necessary permission to access a route. If the user doesn’t have the required permission, they’re redirected to a "No Permission" page.

    NO Access Page

    To define the /no-access route in your app, you can add a Route for the "No Access" page within your AppRoutes component, similar to how you define other routes. This page will be shown if the user doesn’t have the required permissions to access certain routes.

    Steps
    1. Create a NoAccess Page Component: First, create a simple component that will display the "No Access" message.
    2. Add the NoAccess Route: Include the /no-access route in your routing setup (AppRoutes).

    1. Create the NoAccess Component

    Let’s start by creating a simple NoAccess component.

    NoAccess.js (e.g., src/components/NoAccess.js):

    import React from "react";

const NoAccess = () => {
  return (
    <div>
      <h1>No Access</h1>
      <p>You do not have permission to view this page.</p>
    </div>
  );
};

export default NoAccess;

    2. Add the NoAccess Route to Your AppRoutes

    Now, in your AppRoutes component, you need to define a Route that points to the NoAccess page when the user is redirected to /no-access.

    AppRoutes.js
    import React from "react";
import { BrowserRouter as Router, Routes, Route } from "react-router-dom";
import { useSelector } from "react-redux";
import { APP_ROUTES } from "@shared/routes";

// Import components
import PrivateRoute from "./PrivateRoutes";
import RedirectRoute from "./RedirectRoutes";

import Login from "@features/auth/components/Login/Login";
import Dashboard from "@features/dashboard/components/Dashboard";
import MainLayout from "@shared/components/DashbordLayout/MainLayout";
import Analysis from "@features/dashboard/components/Analysis";
import WorkSpace from "@features/dashboard/components/Workspace";
import NotFound from "@shared/components/NotFoundPage";
import BuyingSelling from "@features/forex/components/BuyingSelling";
import AllDeals from "@features/forex/components/AllDeals";
import Users from "@features/settings/components/Users";
import Country from "@features/settings/components/Country";
import NoAccess from "@shared/components/NoAccess"; // Import the NoAccess component

const AppRoutes = () => {
  const isAuthenticated = useSelector((state) => state.auth.isAuthenticated);

  const protectedRoutes = [
    { path: APP_ROUTES.DASHBOARD_ANALYSIS, element: <Analysis /> },
    { path: APP_ROUTES.DASHBOARD_WORKSPACE, element: <WorkSpace /> },
    { path: APP_ROUTES.FOREX_DEALS, element: <AllDeals /> },
    { path: APP_ROUTES.FOREX_TRADING, element: <BuyingSelling /> },
    { path: APP_ROUTES.SETTING_COUNTRY, element: <Country /> },
    { path: APP_ROUTES.SETTING_USERS, element: <Users /> },
    { path: "*", element: <NotFound /> },
  ];

  return (
    <Router>
      <Routes>
        <Route
          path="/"
          element={
            <RedirectRoute>
              <Login />
            </RedirectRoute>
          }
        ></Route>

        {protectedRoutes.map(({ path, element }) => (
          <Route
            key={path}
            path={path}
            element={
              <PrivateRoute requiredPermissions={["CAN_VIEW_DASHBOARD"]}>
                <MainLayout>{element}</MainLayout>
              </PrivateRoute>
            }
          />
        ))}

        {/* No Access Route */}
        <Route path="/no-access" element={<NoAccess />} />

        <Route path="*" element={<NotFound />} />
      </Routes>
    </Router>
  );
};

export default AppRoutes;
